04 וַיְהִ֥י בְּשׁ֖וּב יֵשׁ֑וּעַ וַיְקַבֵּ֤ל אֹתוֹ֙ הָעָ֔ם כִּ֧י כֻלָּ֛ם הָי֥וּ מְחַכִּ֖ים לֽוֹ: 14 וְהִנֵּ֣ה אִישׁ֘ וּשְׁמ֣וֹ יָאִיר֒ וְה֥וּא רֹ֖אשׁ הַכְּנֵ֑סֶת וַיָּבֹ֤א וַיִּפֹּל֙ לְרַגְלֵ֣י יֵשׁ֔וּעַ וַיִּתְחַנֶּן־ל֥וֹ לָב֖וֹא אִתּ֥וֹ אֶל־בֵּיתֽוֹ: 24 כִּ֣י בַּ֛ת יְחִידָ֛ה ל֤וֹ כְּבַת֙ שְׁתֵּים־עֶשְׂרֵ֣ה שָׁנָ֔ה וְהִ֥יא נָטְתָ֖ה לָמ֑וּת וַיְהִ֣י ׀ בְּלֶכְתּ֣וֹ שָׁ֗מָּה וַיִּדְחָקֻ֖הוּ הֲמ֥וֹן הָעָֽם: 34 וְאִשָּׁ֨ה זָבַ֥ת דָּ֛ם שְׁתֵּ֥ים עֶשְׂרֵ֖ה שָׁנָ֑ה אֲשֶׁ֣ר הוֹצִיאָ֤ה כָל־מִחְיָתָהּ֙ לָרֹפְאִ֔ים וְאֵ֥ין אִ֖ישׁ יָכֹ֥ל לְרַפֹּאתָֽהּ: 44 הִ֣יא קָרְבָה֘ מֵאַחֲרָיו֒ וַתִּגַּ֖ע בִּכְנַ֥ף בִּגְד֑וֹ וַיַּעֲמֹ֥ד ז֖וֹב דָּמֶ֥יהָ פִּתְאֹֽם: 54 וַיֹּ֖אמֶר יֵשׁ֑וּעַ מִי־זֶה֙ נָגַע־בִּ֔י וַיְכַחֲשׁוּ֙ כֻלָּ֔ם וַיֹּ֨אמֶר פֶּ֤טְרוֹס וְהָעֹמְדִים֙ אֶצְל֔וֹ מוֹרֶ֞ה הֲמ֣וֹן הָעָ֨ם דֹּחֲקִ֤ים וְלֹחֲצִים֙ אֹתָ֔ךְ וְאַתָּה֙ תֹאמַ֔ר מִ֥י נָגַ֖ע בִּֽי: 64 וַיֹּ֖אמֶר יֵשׁ֑וּעַ נָגַ֤ע בִּי֙ אָדָ֔ם כִּ֣י יָדַ֗עְתִּי אֲשֶׁ֥ר יָצְאָ֖ה מִמֶּ֥נִּי גְּבוּרָֽה: 74 וַתֵּ֧רֶא הָאִשָּׁ֛ה כִּ֣י לֹא־נִסְתְּרָה֙ מִמֶּ֔נּוּ וַתִּגַּ֥שׁ בַּחֲרָדָ֖ה וַתִּפֹּ֣ל לְפָנָ֑יו וַתַּגֵּ֨ד בְּאָזְנֵ֣י כָל־הָעָ֗ם עַל־מֶ֤ה נָגְעָה֙ בּ֔וֹ וְאֵ֥ת אֲשֶׁ֖ר נִרְפְּאָ֥ה פִּתְאֹֽם: 84 וַיֹּ֖אמֶר אֵלֶ֑יהָ {חִזְקִ֨י} בִּתִּ֜י אֱמוּנָתֵךְ֙ הוֹשִׁ֔יעָה לָּ֖ךְ לְכִ֥י לְשָׁלֽוֹם: 94 עוֹדֶ֘נּוּ֘ מְדַבֵּר֒ וְאִ֥ישׁ בָּ֛א מִבֵּ֖ית רֹ֣אשׁ הַכְּנֵ֑סֶת וַיֹּאמַ֕ר מֵתָה֙ בִתְּךָ֔ אַל־תַּטְרִיחַ֖ אֶת־הַמּוֹרֶֽה: 05 וַיִּשְׁמַע֘ יֵשׁ֒וּעַ֒ וַיַּ֖עַן וַיֹּ֑אמֶר ל֚וֹ אַל־תִּירָ֔א רַק־הַאֲמֵ֕ן וְהִ֥יא תִוָּשֵֽׁעַ: 15 וַיָּבֹ֖א הַבַּ֑יְתָה וְלֹא־הִנִּ֨יחַ לְאִ֤ישׁ לָבוֹא֙ אִתּ֔וֹ בִּלְתִּ֧י לְפֶטְר֛וֹס וּלְיַעֲקֹ֣ב וּלְיוֹחָנָ֗ן וְלַאֲבִ֥י הַיַּלְדָּ֖ה וּלְאִמָּֽהּ: 25 וְכֻלָּ֧ם בֹּכִ֛ים וְסֹפְדִ֖ים לָ֑הּ וַיֹּ֕אמֶר אַל־תִּבְכּ֞וּ כִּ֣י ׀ לֹ֣א מֵתָ֗ה אַךְ־יְשֵׁנָ֖ה הִֽיא: 35 וַיִּשְׂחֲק֖וּ עָלָ֑יו בַּאֲשֶׁ֥ר יָדְע֖וּ כִּי־מֵתָֽה: 45 וְה֣וּא אָחַז֘ בְּיָדָהּ֒ וַיִּקְרָ֖א וַיֹּאמַ֑ר הַיַּלְדָּ֖ה קֽוּמִי: 55 וַתָּ֨שָׁב֙ רוּחָ֔הּ וַתָּ֖קָם פִּתְאֹ֑ם וַיְצַ֖ו לָתֶת־לָ֥הּ לֶאֱכֽוֹל: 65 וַיִּתְמְה֥וּ אָבִ֖יהָ
